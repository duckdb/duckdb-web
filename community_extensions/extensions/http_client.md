    WITH __input AS (
      SELECT
        http_get(
            'https://httpbin.org/delay/0',
            headers => MAP {
              'accept': 'application/json',
            },
            params => MAP {
              'limit': 1
            }
        ) AS res
    ),
    __response AS (
      SELECT
        (res->>'status')::INT AS status,
        (res->>'reason') AS reason,
        unnest( from_json(((res->>'body')::JSON)->'headers', '{"Host": "VARCHAR"}') ) AS features

        __input
